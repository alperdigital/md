<script>
    let allKararlar = [];
    let filteredKararlar = [];

    // KararlarÄ± yÃ¼kle
    async function loadKararlar() {
        try {
            // Ã–nce assets/kararlar.json'dan yÃ¼kle
            const response = await fetch('/assets/kararlar.json');
            if (response.ok) {
                allKararlar = await response.json();
            } else {
                // Fallback: _data/kararlar.json'dan dene
                const response2 = await fetch('/_data/kararlar.json');
                if (response2.ok) {
                    allKararlar = await response2.json();
                } else {
                    // Son fallback: GitHub'dan veya dosyalardan yÃ¼kle
                    allKararlar = await loadKararlarFromFiles();
                }
            }
            
            // Mahkeme listesini gÃ¼ncelle
            updateMahkemeFilter();
            
            // Ä°lk filtreleme
            applyFilters();
        } catch (error) {
            console.error('Kararlar yÃ¼klenirken hata:', error);
            allKararlar = await loadKararlarFromFiles();
            updateMahkemeFilter();
            applyFilters();
        }
    }

    // Dosyalardan kararlarÄ± yÃ¼kle (fallback)
    async function loadKararlarFromFiles() {
        // Ã–rnek kararlar
        return [
            {
                kararNo: "2024/12345",
                davaTuru: "Ticaret",
                tarih: "2024-12-15T10:00:00",
                mahkeme: "11. Hukuk Dairesi",
                onemDerecesi: 95,
                baslik: "Ticari Ä°ÅŸletme Devri ve Sorumluluk",
                ozet: "Ticari iÅŸletmenin devri halinde devralanÄ±n sorumluluÄŸu ve kapsamÄ± hakkÄ±nda Ã¶nemli bir karar.",
                icerik: "Bu karar, ticari iÅŸletmenin devri durumunda devralanÄ±n sorumluluÄŸunu ve kapsamÄ±nÄ± detaylÄ± olarak ele almaktadÄ±r."
            }
        ];
    }

    // Mahkeme filtresini gÃ¼ncelle
    function updateMahkemeFilter() {
        const mahkemeler = [...new Set(allKararlar.map(k => k.mahkeme))].sort();
        const select = document.getElementById('filterMahkeme');
        select.innerHTML = '<option value="all">TÃ¼mÃ¼</option>';
        mahkemeler.forEach(mahkeme => {
            const option = document.createElement('option');
            option.value = mahkeme;
            option.textContent = mahkeme;
            select.appendChild(option);
        });
    }

    // Filtreleri uygula
    function applyFilters() {
        const davaTuru = document.getElementById('filterDavaTuru').value;
        const mahkeme = document.getElementById('filterMahkeme').value;
        const sortBy = document.getElementById('sortBy').value;

        filteredKararlar = allKararlar.filter(karar => {
            const davaTuruMatch = davaTuru === 'all' || karar.davaTuru === davaTuru;
            const mahkemeMatch = mahkeme === 'all' || karar.mahkeme === mahkeme;
            return davaTuruMatch && mahkemeMatch;
        });

        // SÄ±rala
        filteredKararlar.sort((a, b) => {
            switch (sortBy) {
                case 'tarih-desc':
                    return new Date(b.tarih) - new Date(a.tarih);
                case 'tarih-asc':
                    return new Date(a.tarih) - new Date(b.tarih);
                case 'onem-desc':
                    return (b.onemDerecesi || 0) - (a.onemDerecesi || 0);
                case 'onem-asc':
                    return (a.onemDerecesi || 0) - (b.onemDerecesi || 0);
                case 'kararNo-asc':
                    return (a.kararNo || '').localeCompare(b.kararNo || '');
                case 'kararNo-desc':
                    return (b.kararNo || '').localeCompare(a.kararNo || '');
                default:
                    return 0;
            }
        });

        renderKararlar();
    }

    // KararlarÄ± render et
    function renderKararlar() {
        const container = document.getElementById('kararlarContainer');
        
        if (filteredKararlar.length === 0) {
            container.innerHTML = '<div class="no-kararlar"><p>ğŸ“­ Filtre kriterlerinize uygun karar bulunamadÄ±.</p></div>';
            return;
        }

        container.innerHTML = filteredKararlar.map(karar => {
            const date = new Date(karar.tarih);
            const dateStr = date.toLocaleDateString('tr-TR', { 
                year: 'numeric', 
                month: 'long', 
                day: 'numeric'
            });

            return `
                <article class="karar-card" onclick="openKararModal(${JSON.stringify(karar).replace(/"/g, '&quot;')})">
                    <div class="karar-card-header">
                        <span class="karar-dava-turu">${karar.davaTuru || 'Genel'}</span>
                        <span class="karar-onem">â­ ${karar.onemDerecesi || 0}</span>
                    </div>
                    <div class="karar-no">ğŸ“‹ Karar No: ${karar.kararNo || 'BelirtilmemiÅŸ'}</div>
                    <h2 class="karar-title">${karar.baslik || 'BaÅŸlÄ±ksÄ±z'}</h2>
                    <p class="karar-ozet">${karar.ozet || ''}</p>
                    <div class="karar-meta">
                        <span class="karar-mahkeme">ğŸ›ï¸ ${karar.mahkeme || 'Mahkeme'}</span>
                        <span class="karar-date">ğŸ“… ${dateStr}</span>
                    </div>
                </article>
            `;
        }).join('');
    }

    // Filtreleri sÄ±fÄ±rla
    function resetFilters() {
        document.getElementById('filterDavaTuru').value = 'all';
        document.getElementById('filterMahkeme').value = 'all';
        document.getElementById('sortBy').value = 'tarih-desc';
        applyFilters();
    }

    // Karar detay modalÄ±nÄ± aÃ§
    function openKararModal(karar) {
        const modal = document.getElementById('kararModal');
        const content = document.getElementById('modalContent');
        
        const date = new Date(karar.tarih);
        const dateStr = date.toLocaleDateString('tr-TR', { 
            year: 'numeric', 
            month: 'long', 
            day: 'numeric'
        });

        content.innerHTML = `
            <div class="karar-detail">
                <div class="karar-detail-header">
                    <span class="karar-dava-turu">${karar.davaTuru || 'Genel'}</span>
                    <span class="karar-onem">â­ ${karar.onemDerecesi || 0}</span>
                </div>
                <div class="karar-detail-no">ğŸ“‹ Karar No: ${karar.kararNo || 'BelirtilmemiÅŸ'}</div>
                <h1>${karar.baslik || 'BaÅŸlÄ±ksÄ±z'}</h1>
                <div class="karar-detail-meta">
                    <span>ğŸ›ï¸ ${karar.mahkeme || 'Mahkeme'}</span>
                    <span>ğŸ“… ${dateStr}</span>
                </div>
                <div class="karar-detail-content">
                    ${karar.icerik || karar.ozet || '<p>Ä°Ã§erik yÃ¼kleniyor...</p>'}
                </div>
            </div>
        `;
        
        modal.style.display = 'block';
    }

    // ModalÄ± kapat
    function closeKararModal() {
        document.getElementById('kararModal').style.display = 'none';
    }

    // Modal dÄ±ÅŸÄ±na tÄ±klanÄ±nca kapat
    window.onclick = function(event) {
        const modal = document.getElementById('kararModal');
        if (event.target === modal) {
            closeKararModal();
        }
    }

    // Event listeners
    document.getElementById('filterDavaTuru').addEventListener('change', applyFilters);
    document.getElementById('filterMahkeme').addEventListener('change', applyFilters);
    document.getElementById('sortBy').addEventListener('change', applyFilters);

    // Sayfa yÃ¼klendiÄŸinde kararlarÄ± yÃ¼kle
    document.addEventListener('DOMContentLoaded', loadKararlar);
</script>
