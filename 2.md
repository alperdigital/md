    // Firebase'den haberleri y√ºkle
    async function loadHaberlerFromFirebase() {
        if (!USE_FIREBASE || typeof firebase === 'undefined') {
            return null;
        }
        
        try {
            // Firebase'i ba≈ülat
            if (!firebase.apps.length) {
                firebase.initializeApp(FIREBASE_CONFIG);
            }
            const database = firebase.database();
            
            // Firebase'den haberleri oku
            const snapshot = await database.ref('haberler').once('value');
            const haberler = snapshot.val();
            
            if (haberler && Array.isArray(haberler)) {
                return haberler;
            } else if (haberler && typeof haberler === 'object') {
                // Eƒüer object ise array'e √ßevir
                return Object.values(haberler);
            }
            return [];
        } catch (error) {
            console.error('Firebase okuma hatasƒ±:', error);
            return null;
        }
    }

    // LocalStorage'dan haberleri y√ºkle
    function loadHaberlerFromLocalStorage() {
        try {
            const stored = localStorage.getItem('haberler_local');
            if (stored) {
                return JSON.parse(stored);
            }
        } catch (e) {
            console.error('LocalStorage okuma hatasƒ±:', e);
        }
        return [];
    }

            if (USE_FIREBASE) {

                    if (response2.ok) {
                        allHaberler = await response2.json();
                    } else {

    async function loadHaberlerFromFiles() {

    function updateAuthorFilter() {

    // Filtreleri uygula
    function applyFilters() {
        const category = document.getElementById('filterCategory').value;
        const author = document.getElementById('filterAuthor').value;
        const sortBy = document.getElementById('sortBy').value;

        filteredHaberler = allHaberler.filter(haber => {
            const categoryMatch = category === 'all' || haber.category === category;
            const authorMatch = author === 'all' || haber.author === author;
            return categoryMatch && authorMatch;
        });

        // Sƒ±rala
        filteredHaberler.sort((a, b) => {
            switch (sortBy) {
                case 'date-desc':
                    return new Date(b.date) - new Date(a.date);
                case 'date-asc':
                    return new Date(a.date) - new Date(b.date);
                case 'popularity-desc':
                    return (b.popularity || 0) - (a.popularity || 0);
                case 'popularity-asc':
                    return (a.popularity || 0) - (b.popularity || 0);
                case 'author-asc':
                    return (a.author || '').localeCompare(b.author || '');
                case 'author-desc':
                    return (b.author || '').localeCompare(a.author || '');
                default:
                    return 0;
            }
        });

        renderHaberler();
    }

    function shareHaber(platform, title, text) {
        // URL'yi temizle (decode edilmi≈ü hali)
        const currentUrl = window.location.origin + window.location.pathname;
        const shareUrl = encodeURIComponent(currentUrl);
        
        // Ba≈ülƒ±k ve metni decode edip tekrar encode et
        const decodedTitle = decodeURIComponent(title);
        const decodedText = decodeURIComponent(text);
        const shareTitle = encodeURIComponent(decodedTitle);
        const shareText = encodeURIComponent(decodedText);
        
        // Payla≈üƒ±m mesajƒ± olu≈ütur
        const shareMessage = `${decodedTitle} - ${decodedText} ${currentUrl}`;
        const encodedMessage = encodeURIComponent(shareMessage);
        
        let shareLink = '';
        
        switch(platform) {
            case 'facebook':
                shareLink = `https://www.facebook.com/sharer/sharer.php?u=${shareUrl}&quote=${shareTitle}`;
                break;
            case 'twitter':
                shareLink = `https://twitter.com/intent/tweet?url=${shareUrl}&text=${shareTitle}%20-%20${shareText}`;
                break;
            case 'whatsapp':
                shareLink = `https://wa.me/?text=${encodedMessage}`;
                break;
            case 'telegram':
                shareLink = `https://t.me/share/url?url=${shareUrl}&text=${shareTitle}`;
                break;
            case 'linkedin':
                shareLink = `https://www.linkedin.com/sharing/share-offsite/?url=${shareUrl}`;
                break;
            case 'email':
                shareLink = `mailto:?subject=${shareTitle}&body=${shareText}%20${shareUrl}`;
