    // Haberleri y√ºkle
    async function loadHaberler() {
        try {
            // √ñnce assets/haberler.json'dan y√ºkle
            const response = await fetch('/assets/haberler.json');
            if (response.ok) {
                allHaberler = await response.json();
            } else {
                // Fallback: _data/haberler.json'dan dene
                const response2 = await fetch('/_data/haberler.json');
                if (response2.ok) {
                    allHaberler = await response2.json();
                } else {
                    // Son fallback: GitHub'dan veya dosyalardan y√ºkle
                    allHaberler = await loadHaberlerFromFiles();
                }
            }
            
            // Yazar listesini g√ºncelle
            updateAuthorFilter();
            
            // ƒ∞lk filtreleme
            applyFilters();
        } catch (error) {
            console.error('Haberler y√ºklenirken hata:', error);
            try {
                allHaberler = await loadHaberlerFromFiles();
                updateAuthorFilter();
                applyFilters();
            } catch (fallbackError) {
                console.error('Fallback y√ºkleme hatasƒ±:', fallbackError);
                document.getElementById('haberlerContainer').innerHTML = '<div class="no-haberler"><p>üì≠ Haberler y√ºklenirken bir sorun olu≈ütu. L√ºtfen sayfayƒ± yenileyin.</p></div>';
            }
        }
    }

    // Dosyalardan haberleri y√ºkle (fallback)
    async function loadHaberlerFromFiles() {
        const haberler = [];
        const haberFiles = [
            { file: '2025-01-16-ornek-haber-1.md', url: '/haberler/2025-01-16-ornek-haber-1/' },
            { file: '2025-01-15-ornek-haber-2.md', url: '/haberler/2025-01-15-ornek-haber-2/' }
        ];

        for (const haberFile of haberFiles) {
            try {
                const response = await fetch(haberFile.url);
                if (response.ok) {
                    const html = await response.text();
                    const parser = new DOMParser();
                    const doc = parser.parseFromString(html, 'text/html');
                    
                    // Front matter'den bilgileri √ßƒ±kar (basit parse)
                    const haber = {
                        title: doc.querySelector('h1')?.textContent || 'Ba≈ülƒ±ksƒ±z',
                        date: new Date(haberFile.file.substring(0, 10)),
                        author: 'Yazar',
                        category: 'G√ºndem',
                        popularity: Math.floor(Math.random() * 100),
                        excerpt: doc.querySelector('p')?.textContent || '',
                        url: haberFile.url,
                        content: doc.querySelector('.page-content')?.innerHTML || ''
                    };
                    haberler.push(haber);
                }
            } catch (e) {
                console.warn('Haber y√ºklenemedi:', haberFile.file);
            }
        }

        return haberler;
    }

    // Yazar filtresini g√ºncelle
    function updateAuthorFilter() {
        const authors = [...new Set(allHaberler.map(h => h.author))].sort();
        const select = document.getElementById('filterAuthor');
        select.innerHTML = '<option value="all">T√ºm√º</option>';
        authors.forEach(author => {
            const option = document.createElement('option');
            option.value = author;
            option.textContent = author;
            select.appendChild(option);
        });
    }

    // Filtreleri uygula
    function applyFilters() {
        const category = document.getElementById('filterCategory').value;
        const author = document.getElementById('filterAuthor').value;
        const sortBy = document.getElementById('sortBy').value;

        filteredHaberler = allHaberler.filter(haber => {
            const categoryMatch = category === 'all' || haber.category === category;
            const authorMatch = author === 'all' || haber.author === author;
            return categoryMatch && authorMatch;
        });

        // Sƒ±rala
        filteredHaberler.sort((a, b) => {
            switch (sortBy) {
                case 'date-desc':
                    return new Date(b.date) - new Date(a.date);
                case 'date-asc':
                    return new Date(a.date) - new Date(b.date);
                case 'popularity-desc':
                    return (b.popularity || 0) - (a.popularity || 0);
                case 'popularity-asc':
                    return (a.popularity || 0) - (b.popularity || 0);
                case 'author-asc':
                    return (a.author || '').localeCompare(b.author || '');
                case 'author-desc':
                    return (b.author || '').localeCompare(a.author || '');
                default:
                    return 0;
            }
        });

        renderHaberler();
    }

    // Sosyal medya payla≈üƒ±m fonksiyonu
    function shareHaber(platform, title, text) {
        const currentUrl = window.location.origin + window.location.pathname;
        const shareUrl = encodeURIComponent(currentUrl);
        const shareTitle = encodeURIComponent(title);
        const shareText = encodeURIComponent(text);
        
        let shareLink = '';
        
        switch(platform) {
            case 'facebook':
                shareLink = `https://www.facebook.com/sharer/sharer.php?u=${shareUrl}&quote=${shareTitle}`;
                break;
            case 'twitter':
                shareLink = `https://twitter.com/intent/tweet?url=${shareUrl}&text=${shareTitle}%20${shareText}`;
                break;
            case 'whatsapp':
                shareLink = `https://wa.me/?text=${shareTitle}%20${shareUrl}`;
                break;
            case 'telegram':
                shareLink = `https://t.me/share/url?url=${shareUrl}&text=${shareTitle}`;
                break;
            case 'linkedin':
                shareLink = `https://www.linkedin.com/sharing/share-offsite/?url=${shareUrl}`;
                break;
            case 'email':
                shareLink = `mailto:?subject=${shareTitle}&body=${shareText}%20${shareUrl}`;
                break;
            default:
                return;
        }
        
        // Yeni pencerede a√ß (mobilde uygulama a√ßƒ±lƒ±r)
        if (platform === 'email') {
            window.location.href = shareLink;
        } else {
            window.open(shareLink, '_blank', 'width=600,height=400');
        }
    }
