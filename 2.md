<script>
    let allHaberler = [];
    let filteredHaberler = [];

    // Haberleri y√ºkle
    async function loadHaberler() {
        try {
            // √ñnce assets/haberler.json'dan y√ºkle
            const response = await fetch('/assets/haberler.json');
            if (response.ok) {
                allHaberler = await response.json();
            } else {
                // Fallback: _data/haberler.json'dan dene
                const response2 = await fetch('/_data/haberler.json');
                if (response2.ok) {
                    allHaberler = await response2.json();
                } else {
                    // Son fallback: GitHub'dan veya dosyalardan y√ºkle
                    allHaberler = await loadHaberlerFromFiles();
                }
            }
            
            // Yazar listesini g√ºncelle
            updateAuthorFilter();
            
            // ƒ∞lk filtreleme
            applyFilters();
        } catch (error) {
            console.error('Haberler y√ºklenirken hata:', error);
            try {
                allHaberler = await loadHaberlerFromFiles();
                updateAuthorFilter();
                applyFilters();
            } catch (fallbackError) {
                console.error('Fallback y√ºkleme hatasƒ±:', fallbackError);
                document.getElementById('haberlerContainer').innerHTML = '<div class="no-haberler"><p>üì≠ Haberler y√ºklenirken bir sorun olu≈ütu. L√ºtfen sayfayƒ± yenileyin.</p></div>';
            }
        }
    }

    // Dosyalardan haberleri y√ºkle (fallback)
    async function loadHaberlerFromFiles() {
        const haberler = [];
        const haberFiles = [
            { file: '2025-01-16-ornek-haber-1.md', url: '/haberler/2025-01-16-ornek-haber-1/' },
            { file: '2025-01-15-ornek-haber-2.md', url: '/haberler/2025-01-15-ornek-haber-2/' }
        ];

        for (const haberFile of haberFiles) {
            try {
                const response = await fetch(haberFile.url);
                if (response.ok) {
                    const html = await response.text();
                    const parser = new DOMParser();
                    const doc = parser.parseFromString(html, 'text/html');
                    
                    // Front matter'den bilgileri √ßƒ±kar (basit parse)
                    const haber = {
                        title: doc.querySelector('h1')?.textContent || 'Ba≈ülƒ±ksƒ±z',
                        date: new Date(haberFile.file.substring(0, 10)),
                        author: 'Yazar',
                        category: 'G√ºndem',
                        popularity: Math.floor(Math.random() * 100),
                        excerpt: doc.querySelector('p')?.textContent || '',
                        url: haberFile.url,
                        content: doc.querySelector('.page-content')?.innerHTML || ''
                    };
                    haberler.push(haber);
                }
            } catch (e) {
                console.warn('Haber y√ºklenemedi:', haberFile.file);
            }
        }

        return haberler;
    }

    // Yazar filtresini g√ºncelle
    function updateAuthorFilter() {
        const authors = [...new Set(allHaberler.map(h => h.author))].sort();
        const select = document.getElementById('filterAuthor');
        select.innerHTML = '<option value="all">T√ºm√º</option>';
        authors.forEach(author => {
            const option = document.createElement('option');
            option.value = author;
            option.textContent = author;
            select.appendChild(option);
        });
    }

    // Filtreleri uygula
    function applyFilters() {
        const category = document.getElementById('filterCategory').value;
        const author = document.getElementById('filterAuthor').value;
        const sortBy = document.getElementById('sortBy').value;

        filteredHaberler = allHaberler.filter(haber => {
            const categoryMatch = category === 'all' || haber.category === category;
            const authorMatch = author === 'all' || haber.author === author;
            return categoryMatch && authorMatch;
        });

        // Sƒ±rala
        filteredHaberler.sort((a, b) => {
            switch (sortBy) {
                case 'date-desc':
                    return new Date(b.date) - new Date(a.date);
                case 'date-asc':
                    return new Date(a.date) - new Date(b.date);
                case 'popularity-desc':
                    return (b.popularity || 0) - (a.popularity || 0);
                case 'popularity-asc':
                    return (a.popularity || 0) - (b.popularity || 0);
                case 'author-asc':
                    return (a.author || '').localeCompare(b.author || '');
                case 'author-desc':
                    return (b.author || '').localeCompare(a.author || '');
                default:
                    return 0;
            }
        });

        renderHaberler();
    }

    // Haberleri render et
    function renderHaberler() {
        const container = document.getElementById('haberlerContainer');
        
        if (filteredHaberler.length === 0) {
            container.innerHTML = '<div class="no-haberler"><p>üì≠ Filtre kriterlerinize uygun haber bulunamadƒ±.</p></div>';
            return;
        }

        container.innerHTML = filteredHaberler.map(haber => {
            const date = new Date(haber.date);
            const dateStr = date.toLocaleDateString('tr-TR', { 
                year: 'numeric', 
                month: 'long', 
                day: 'numeric',
                hour: '2-digit',
                minute: '2-digit'
            });

            return `
                <article class="haber-card" onclick="openHaberModal('${haber.url || ''}', ${JSON.stringify(haber).replace(/"/g, '&quot;').replace(/'/g, '&#39;')})">
                    <div class="haber-card-header">
                        <span class="haber-category">${haber.category || 'Genel'}</span>
                        <span class="haber-popularity">üî• ${haber.popularity || 0}</span>
                    </div>
                    <h2 class="haber-title">${haber.title || 'Ba≈ülƒ±ksƒ±z'}</h2>
                    <p class="haber-excerpt">${haber.excerpt || ''}</p>
                    <div class="haber-meta">
                        <span class="haber-author">‚úçÔ∏è ${haber.author || 'Yazar'}</span>
                        <span class="haber-date">üìÖ ${dateStr}</span>
                    </div>
                </article>
            `;
        }).join('');
    }

    // Filtreleri sƒ±fƒ±rla
    function resetFilters() {
        document.getElementById('filterCategory').value = 'all';
        document.getElementById('filterAuthor').value = 'all';
        document.getElementById('sortBy').value = 'date-desc';
        applyFilters();
    }

    // Haber detay modalƒ±nƒ± a√ß
    function openHaberModal(url, haberStr) {
        try {
            // Eƒüer haber string ise parse et
            let haber;
            if (typeof haberStr === 'string') {
                haber = JSON.parse(haberStr.replace(/&quot;/g, '"'));
            } else {
                haber = haberStr;
            }
            
            const modal = document.getElementById('haberModal');
            const content = document.getElementById('modalContent');
            
            if (!modal || !content) {
                console.error('Modal elementleri bulunamadƒ±');
                return;
            }
            
            const date = new Date(haber.date || new Date());
            const dateStr = date.toLocaleDateString('tr-TR', { 
                year: 'numeric', 
                month: 'long', 
                day: 'numeric',
                hour: '2-digit',
                minute: '2-digit'
            });

            // XSS korumasƒ± i√ßin escape
            function escapeHTML(text) {
                const div = document.createElement('div');
                div.textContent = text;
                return div.innerHTML;
            }

            content.innerHTML = `
                <div class="haber-detail">
                    <div class="haber-detail-header">
                        <span class="haber-category">${escapeHTML(haber.category || 'Genel')}</span>
                        <span class="haber-popularity">üî• ${haber.popularity || 0}</span>
                    </div>
                    <h1>${escapeHTML(haber.title || 'Ba≈ülƒ±ksƒ±z')}</h1>
                    <div class="haber-detail-meta">
                        <span>‚úçÔ∏è ${escapeHTML(haber.author || 'Yazar')}</span>
                        <span>üìÖ ${dateStr}</span>
                    </div>
                    <div class="haber-detail-content">
                        ${haber.content ? escapeHTML(haber.content) : (haber.excerpt ? escapeHTML(haber.excerpt) : '<p>ƒ∞√ßerik y√ºkleniyor...</p>')}
                    </div>
                </div>
            `;
            
            modal.style.display = 'block';
        } catch (error) {
            console.error('Modal a√ßƒ±lƒ±rken hata:', error);
            alert('Haber detaylarƒ± y√ºklenirken bir hata olu≈ütu.');
        }
    }

    // Modalƒ± kapat
    function closeHaberModal() {
        document.getElementById('haberModal').style.display = 'none';
    }

    // Modal dƒ±≈üƒ±na tƒ±klanƒ±nca kapat
    window.onclick = function(event) {
        const modal = document.getElementById('haberModal');
        if (event.target === modal) {
            closeHaberModal();
        }
    }

    // Event listeners
    document.getElementById('filterCategory').addEventListener('change', applyFilters);
    document.getElementById('filterAuthor').addEventListener('change', applyFilters);
    document.getElementById('sortBy').addEventListener('change', applyFilters);

    // Sayfa y√ºklendiƒüinde haberleri y√ºkle
    document.addEventListener('DOMContentLoaded', loadHaberler);
</script>
